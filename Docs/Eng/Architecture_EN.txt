================================================================================
GAMMA ENGINE: PROJECT ARCHITECTURE 4.0
================================================================================

ROOT DIRECTORY
|-- GammaEngine.h/cpp       # Engine Core. Orchestrator of subsystem lifecycles.
|-- Main.cpp                # Entry point. Initialization of base services.

SRC/CONFIG/ (Configuration)
|-- EngineConfig.h/cpp      # Global engine settings and launch parameter parsing.

SRC/CORE/ (Base Systems)
|-- Window.h/cpp            # WinAPI wrapper and window context management.
|-- InputSystem.h/cpp       # Input processing (Action Mapping, Raw Input).
|-- TaskScheduler.h/cpp     # Multithreading: Task scheduler and Worker Threads.
|-- ResourceManager.h/cpp    # Global resource manager and caching system.
|-- Timer.h/cpp             # High-precision timer with frame smoothing logic.
|-- Logger.h/cpp            # Thread-safe logging (Console + File).
|-- DataSection.h/cpp       # Base interface for working with data structures.
|-- Prerequisites.h         # Pre-definitions, macros, and global types.

SRC/GRAPHICS/ (Rendering & GPU Pipeline)
|-- DX11Renderer.h/cpp      # Low-level API (DirectX 11 Device/Context).
|-- RenderPipeline.h/cpp    # Management of rendering stages and passes.
|-- RenderTarget.h/cpp      # Render buffer management and G-Buffer.
|-- GpuCullingSystem.h/cpp  # GPU Frustum and Occlusion culling.
|-- HzbManager.h/cpp        # Hierarchical Z-Buffer for accelerated culling.
|-- StaticGpuScene.h/cpp    # GPU-driven rendering system for static objects.
|-- TerrainGpuScene.h/cpp   # GPU representation of the landscape.
|-- ComputeShader.h/cpp     # Wrapper for compute shaders.
|-- Camera.h/cpp            # Camera logic, View/Projection matrices, and Frustum.
|-- ShaderManager.h/cpp     # Shader registry and automatic compilation.
|-- ConstantBuffer.h        # Templates for managing constant buffers.
|-- Texture/TextureArray.h  # Handling of individual textures and texture arrays.
|-- GpuTypes.h              # Shared data structures for CPU and GPU.

SRC/WORLD/ (World & Logic)
|-- WorldLoader.h/cpp       # Asynchronous loading of game spaces.
|-- Chunk.h/cpp             # World sector: Object management and LOD.
|-- Terrain.h/cpp           # Landscape visualization and collisions.
|-- WaterVLO.h/cpp          # Water system (Very Large Object).

SRC/RESOURCES/ (Data & Formats)
|-- TerrainBaker.h/cpp      # Offline tool for baking terrain data.
|-- TerrainAtlasManager.h   # Generation and management of terrain texture atlases.
|-- TerrainArrayManager.h   # Management of terrain layer arrays.
|-- BwPackedSection.h/cpp   # Parser for binary XML/KV sections (BigWorld format).
|-- BwBinSection.h/cpp      # Processing of .odata and geometry.
|-- HeightMap.h/cpp         # Decoding of height maps.
|-- SpaceSettings.h/cpp     # Space parameter configuration (space.settings).
|-- StbImageImpl.cpp        # Image loading implementation (stb_image).
|-- tiny_gltf.h / json.hpp  # Support for GLTF and JSON formats.

SRC/GUI/ (Interface)
|-- DebugOverlay.h/cpp      # Performance monitoring and debugging tools.
